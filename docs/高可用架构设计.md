## 2. 熔断降级设计

### 2.1 Resilience4j 配置

```yaml
# application.yml
resilience4j:
  circuitbreaker:
    configs:
      default:
        slidingWindowSize: 10
        failureRateThreshold: 50
        waitDurationInOpenState: 10s
        permittedNumberOfCallsInHalfOpenState: 3
        slowCallDurationThreshold: 2s
        slowCallRateThreshold: 100
    instances:
      embeddingService:
        baseConfig: default
        slowCallDurationThreshold: 5s  # AI服务允许更长的响应时间
      ragService:
        baseConfig: default
        slowCallDurationThreshold: 10s
        
  ratelimiter:
    configs:
      default:
        limitForPeriod: 10
        limitRefreshPeriod: 1s
        timeoutDuration: 0
    instances:
      aiApi:
        limitForPeriod: 5  # AI API 每秒5次调用限制
        limitRefreshPeriod: 1s
```

### 2.2 熔断器使用示例

```java
@Service
public class EmbeddingServiceClient {
    
    private final EmbeddingModel embeddingModel;
    
    @CircuitBreaker(name = "embeddingService", fallbackMethod = "embeddingFallback")
    @RateLimiter(name = "aiApi")
    @Retry(name = "embeddingService")
    public float[] generateEmbedding(String text) {
        return embeddingModel.embed(text);
    }
    
    // 降级方法：返回缓存向量或零向量
    public float[] embeddingFallback(String text, Exception e) {
        log.warn("Embedding服务降级，文本: {}, 错误: {}", text, e.getMessage());
        // 尝试从缓存获取
        float[] cached = cacheService.getCachedEmbedding(text);
        if (cached != null) {
            return cached;
        }
        // 返回空向量表示失败
        return new float[1536]; // 默认维度
    }
}
```

### 2.3 熔断状态监控

```java
@RestController
@RequestMapping("/actuator")
public class CircuitBreakerMonitor {
    
    @GetMapping("/circuit-breakers")
    public Map<String, Object> getCircuitBreakerStatus() {
        CircuitBreakerRegistry registry = CircuitBreakerRegistry.ofDefaults();
        Map<String, Object> status = new HashMap<>();
        
        registry.getAllCircuitBreakers().forEach(cb -> {
            Map<String, Object> cbStatus = new HashMap<>();
            cbStatus.put("state", cb.getState());
            cbStatus.put("failureRate", cb.getMetrics().getFailureRate());
            cbStatus.put("numberOfCalls", cb.getMetrics().getNumberOfCalls());
            status.put(cb.getName(), cbStatus);
        });
        
        return status;
    }
}
```

### 3.1 消息队列架构


### 3.2 RabbitMQ 配置

```java
@Configuration
public class RabbitMQConfig {
    
    public static final String DOCUMENT_QUEUE = "document.process.queue";
    public static final String EMBEDDING_QUEUE = "embedding.process.queue";
    public static final String DLQ = "document.process.dlq";
    
    @Bean
    public Queue documentQueue() {
        return QueueBuilder.durable(DOCUMENT_QUEUE)
                .withArgument("x-dead-letter-exchange", "")
                .withArgument("x-dead-letter-routing-key", DLQ)
                .build();
    }
    
    @Bean
    public Queue embeddingQueue() {
        return QueueBuilder.durable(EMBEDDING_QUEUE)
                .build();
    }
    
    @Bean
    public Queue deadLetterQueue() {
        return new Queue(DLQ, true);
    }
}
```

### 3.3 异步文档处理

```java
@Service
public class AsyncDocumentProcessor {
    
    @RabbitListener(queues = DOCUMENT_QUEUE)
    @Async
    public void processDocument(DocumentMessage message) {
        try {
            // 1. 解析文档
            String content = documentParser.parse(message.getFilePath());
            
            // 2. 分割文本
            List<String> chunks = textSplitter.split(content);
            
            // 3. 发送到嵌入队列
            for (String chunk : chunks) {
                rabbitTemplate.convertAndSend(
                    EMBEDDING_QUEUE, 
                    new EmbeddingMessage(chunk, message.getDocumentId())
                );
            }
            
            log.info("文档处理完成: {}", message.getDocumentId());
        } catch (Exception e) {
            log.error("文档处理失败: {}", e.getMessage());
            throw new AmqpRejectAndDontRequeueException(e);
        }
    }
}
```

## 4. 流式响应设计

### 4.1 SSE (Server-Sent Events) 实现

```java
@RestController
@RequestMapping("/api/v1/rag")
public class StreamingRagController {
    
    private final RagService ragService;
    
    @GetMapping(value = "/chat/stream", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public Flux<ServerSentEvent<String>> streamChat(
            @RequestParam String query,
            @RequestParam(required = false) String conversationId) {
        
        return ragService.streamAnswer(query, conversationId)
                .map(chunk -> ServerSentEvent.<String>builder()
                        .data(chunk)
                        .event("message")
                        .build())
                .concatWith(Flux.just(
                        ServerSentEvent.<String>builder()
                                .data("[DONE]")
                                .event("done")
                                .build()
                ));
    }
}
```

### 4.2 流式RAG服务

```java
@Service
public class StreamingRagService {
    
    private final ChatClient chatClient;
    private final VectorStore vectorStore;
    
    public Flux<String> streamAnswer(String query, String conversationId) {
        // 1. 检索相关文档
        List<Document> documents = vectorStore.similaritySearch(
            SearchRequest.query(query).withTopK(5)
        );
        
        // 2. 构建上下文
        String context = documents.stream()
                .map(Document::getContent)
                .collect(Collectors.joining("\n\n"));
        
        // 3. 构建提示词
        String prompt = buildPrompt(query, context);
        
        // 4. 流式生成答案
        return chatClient.stream()
                .prompt(prompt)
                .stream()
                .content();
    }
}
```

## 5. 缓存策略设计

### 5.1 多级缓存架构

```
┌─────────────────────────────────────────────────────────┐
│                    请求入口                              │
└─────────────────────────┬───────────────────────────────┘
                          │
                          ▼
┌─────────────────────────────────────────────────────────┐
│              L1: 本地缓存 (Caffeine)                     │
│              - 热点数据                                  │
│              - TTL: 5分钟                                │
│              - 最大容量: 1000                            │
└─────────────────────────┬───────────────────────────────┘
                          │ Miss
                          ▼
┌─────────────────────────────────────────────────────────┐
│              L2: 分布式缓存 (Redis)                      │
│              - 共享数据                                  │
│              - TTL: 1小时                                │
│              - 向量缓存                                  │
└─────────────────────────┬───────────────────────────────┘
                          │ Miss
                          ▼
┌─────────────────────────────────────────────────────────┐
│              L3: 向量数据库 (Milvus)                     │
│              - 持久化数据                                │
│              - 向量检索                                  │
└─────────────────────────────────────────────────────────┘
```

### 5.2 缓存配置

```java
@Configuration
public class CacheConfig {
    
    @Bean
    public CacheManager cacheManager(RedisConnectionFactory factory) {
        // L1: Caffeine本地缓存
        CaffeineCacheManager caffeineManager = new CaffeineCacheManager();
        caffeineManager.setCaffeine(Caffeine.newBuilder()
                .maximumSize(1000)
                .expireAfterWrite(5, TimeUnit.MINUTES)
                .recordStats());
        
        // L2: Redis分布式缓存
        RedisCacheManager redisManager = RedisCacheManager.builder(factory)
                .cacheDefaults(RedisCacheConfiguration.defaultCacheConfig()
                        .entryTtl(Duration.ofHours(1))
                        .serializeValuesWith(RedisSerializationContext.SerializationPair
                                .fromSerializer(new GenericJackson2JsonRedisSerializer())))
                .build();
        
        return new CompositeCacheManager(caffeineManager, redisManager);
    }
}
```

### 6.3 健康检查

```java
@Component
public class MilvusHealthIndicator implements HealthIndicator {
    
    private final MilvusClient milvusClient;
    
    @Override
    public Health health() {
        try {
            // 检查Milvus连接
            var response = milvusClient.hasCollection(
                HasCollectionParam.newBuilder()
                    .withCollectionName("knowledge_base")
                    .build()
            );
            
            if (response.getStatus() == R.Status.Success.getCode()) {
                return Health.up()
                        .withDetail("collection", "knowledge_base")
                        .build();
            }
            
            return Health.down()
                    .withDetail("error", "Collection not found")
                    .build();
        } catch (Exception e) {
            return Health.down(e).build();
        }
    }
}
```

## 8. 容灾与恢复

### 8.1 数据备份策略

| 数据类型 | 备份频率 | 保留时间 |
|----------|----------|----------|
| 向量数据 | 每日增量 | 30天 |
| 配置数据 | 实时同步 | 永久 |
| 日志数据 | 实时收集 | 7天 |

### 8.2 故障恢复流程

```
1. 检测故障 → 监控系统告警
2. 自动切换 → 熔断器降级
3. 人工介入 → 排查根因
4. 服务恢复 → 重启/回滚
5. 事后复盘 → 优化改进
```
